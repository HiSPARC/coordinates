\pgfmathsetmacro{\alphavec}{90-40}
\pgfmathsetmacro{\rlen}{1.5}
\pgfmathsetmacro{\betavec}{90-120}

\begin{tikzpicture}[scale=2,thick]

    \definecolor{enu}{RGB}{0,184,0}; % green
    \definecolor{compass}{RGB}{125,10,255}; % purple
    \definecolor{base}{RGB}{128,128,128}; % grey

    % Origin
    \coordinate (O) at (0,0);

    % Draw ENU axes
    \draw[enu,->,-stealth] (O) -- (0,2) node[anchor=south] {$North$};
    \draw[enu,->,-stealth] (O) -- (2,0) node[anchor=west] {$East$};

    % Draw a detectors
    \coordinate (d0) at (\alphavec:\rlen);
    \fill[base,rotate=\betavec] (d0) ++(-.15,-.075) rectangle +(.3,.15);

    % Draw distance line
    \draw[compass,->,-stealth] (O) -- node[below] {$r$} (d0);
    % Draw angle line
    \draw[compass,->,-stealth] (0,.5) arc (90:\alphavec:.5);
    \draw[compass] (70:.6) node {$\alpha$};

    % Draw North from detector
    \draw[->,-stealth] (d0) -- +(0,.75);
    % Draw detector angle
    \draw[->,-stealth] (d0) -- +(\betavec:.65);

    % Draw angle line
    \draw[compass,->,-stealth] (d0) ++(0,.4) arc (90:\betavec:.4);
    \draw[compass] (d0) ++(60:.45) node[anchor=west] {$\beta$};

\end{tikzpicture}
